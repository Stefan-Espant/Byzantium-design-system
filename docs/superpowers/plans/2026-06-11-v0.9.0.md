```typescript
import type { Meta, StoryObj } from '@storybook/vue3'

export type StoryMeta<T> = Meta<T>
export type Story<T> = StoryObj<T>

export function defaultStory<T>(args: Partial<T>): Story<T> {
  return { args: args as any }
```

```typescript
import { ByzButton } from '@byzantium-design-system/core'

const meta: Meta<typeof ByzButton> = {
```

```vue
  import { ref, onMounted, onUnmounted } from 'vue'
  defineOptions({ name: 'ByzCountdown' })

  interface Props {
    target: Date
    labels?: { days: string; hours: string; minutes: string; seconds: string }
  }

  const props = withDefaults(defineProps<Props>(), {
    labels: () => ({ days: 'Dagen', hours: 'Uur', minutes: 'Min', seconds: 'Sec' }),
  })

  const emit = defineEmits<{ expired: [] }>()

  interface TimeLeft { days: number; hours: number; minutes: number; seconds: number }

  function calculate(): TimeLeft {
    const diff = props.target.getTime() - Date.now()
    if (diff <= 0) {
      emit('expired')
      return { days: 0, hours: 0, minutes: 0, seconds: 0 }
    }
    return {
      days:    Math.floor(diff / 86_400_000),
      hours:   Math.floor((diff % 86_400_000) / 3_600_000),
      minutes: Math.floor((diff % 3_600_000)  / 60_000),
      seconds: Math.floor((diff % 60_000)     / 1_000),
    }
  }

  const timeLeft = ref<TimeLeft>(calculate())
  let timer: ReturnType<typeof setInterval>

  onMounted(() => { timer = setInterval(() => { timeLeft.value = calculate() }, 1000) })
  onUnmounted(() => clearInterval(timer))
</script>

<template>
  <div class="byz-countdown" role="timer" :aria-label="`Afteller tot ${target.toLocaleDateString()}`">
    <div
      v-for="[key, label] in Object.entries(labels)"
      :key="key"
      data-unit
      class="byz-countdown__unit"
    >
      <span class="byz-countdown__value">{{ String(timeLeft[key as keyof TimeLeft]).padStart(2, '0') }}</span>
      <span class="byz-countdown__label">{{ label }}</span>
    </div>
  </div>
</template>
```

```vue
<script setup lang="ts">
  import { ref } from 'vue'
  import { useLocalStorage } from '../../../composables/useLocalStorage'
  defineOptions({ name: 'ByzCookieBanner' })

  interface Props {
    message?: string
    acceptLabel?: string
    rejectLabel?: string
    preferencesLabel?: string
  }

  withDefaults(defineProps<Props>(), {
    message: 'Wij gebruiken cookies om de ervaring te verbeteren.',
    acceptLabel: 'Accepteren',
    rejectLabel: 'Weigeren',
    preferencesLabel: 'Voorkeuren',
  })

  const emit = defineEmits<{
    accept: []
    reject: []
    preferences: []
  }>()

  const consent = useLocalStorage<string | null>('byz-cookie-consent', null)
  const visible = ref(consent.value === null)

  function accept() {
    consent.value = 'accepted'
    visible.value = false
    emit('accept')
  }

  function reject() {
    consent.value = 'rejected'
    visible.value = false
    emit('reject')
  }
</script>

<template>
  <div
    v-if="visible"
    class="byz-cookie-banner"
    role="dialog"
    aria-label="Cookie-toestemming"
    aria-modal="false"

    <p class="byz-cookie-banner__message">{{ message }}</p>
    <div class="byz-cookie-banner__actions">
      <button type="button" data-action="reject"  class="byz-cookie-banner__btn byz-cookie-banner__btn--secondary" @click="reject">{{ rejectLabel }}</button>
      <button type="button" data-action="preferences" class="byz-cookie-banner__btn byz-cookie-banner__btn--ghost" @click="emit('preferences')">{{ preferencesLabel }}</button>
      <button type="button" data-action="accept" class="byz-cookie-banner__btn byz-cookie-banner__btn--primary" @click="accept">{{ acceptLabel }}</button>
```

```vue
  import { ref } from 'vue'
```
