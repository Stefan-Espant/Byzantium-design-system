```typescript
// packages/byzantium/src/composables/useVirtualList.ts
import { ref, computed, onMounted, onUnmounted, type Ref } from 'vue'

interface VirtualListOptions {
  itemHeight: number
  overscan?: number
}

interface VirtualItem<T> {
  item: T
  index: number
  offsetTop: number
}

export function useVirtualList<T>(
  list: Ref<T[]>,
  options: VirtualListOptions,
) {
  const { itemHeight, overscan = 3 } = options

  const containerRef = ref<HTMLElement | null>(null)
  const scrollTop    = ref(0)
  const containerH   = ref(0)

  const startIndex = computed(() => {
    const idx = Math.floor(scrollTop.value / itemHeight) - overscan
    return Math.max(0, idx)
  })

  const endIndex = computed(() => {
    const idx = Math.ceil((scrollTop.value + containerH.value) / itemHeight) + overscan
    return Math.min(list.value.length - 1, idx)
  })

  const visibleItems = computed<VirtualItem<T>[]>(() => {
    if (containerH.value === 0) return []
    const result: VirtualItem<T>[] = []
    for (let i = startIndex.value; i <= endIndex.value; i++) {
      result.push({ item: list.value[i], index: i, offsetTop: i * itemHeight })
    }
    return result
  })

  const totalHeight = computed(() => list.value.length * itemHeight)

  function onScroll(e: Event) {
    scrollTop.value = (e.target as HTMLElement).scrollTop
  }

  function measure() {
    if (containerRef.value) containerH.value = containerRef.value.clientHeight
  }

  let ro: ResizeObserver | null = null

  onMounted(() => {
    if (!containerRef.value) return
    containerH.value = containerRef.value.clientHeight
    containerRef.value.addEventListener('scroll', onScroll, { passive: true })
    ro = new ResizeObserver(measure)
    ro.observe(containerRef.value)
  })

  onUnmounted(() => {
    containerRef.value?.removeEventListener('scroll', onScroll)
    ro?.disconnect()
  })

  return { containerRef, visibleItems, startIndex, totalHeight }
}
```

```vue
    items: unknown[]
    itemHeight: number

  <div
    :ref="el => { containerRef = el as HTMLElement | null }"
    class="byz-virtual-list"

    :aria-label="ariaLabel"
    :aria-rowcount="items.length"

        :style="{

          top: `${vItem.offsetTop}px`,
          width: '100%',
          height: `${itemHeight}px`,

        role="listitem"
        :aria-rowindex="vItem.index + 1"

      </div>
    </div>
  </div>
```

```vue
<script setup lang="ts">
  import { ref } from 'vue'
  import { useDebounce } from '../../../composables/useDebounce'
  defineOptions({ name: 'ByzFilterBar' })

  interface FilterDef {
    key: string
    label: string
    options: string[]
  }

  interface Props {
    filters: FilterDef[]
    modelValue: Record<string, string>
    placeholder?: string
  }

  const props = withDefaults(defineProps<Props>(), { placeholder: 'Zoeken...' })
  const emit  = defineEmits<{
    'update:modelValue': [v: Record<string, string>]
    search: [query: string]
  }>()

  const searchInput = ref('')
  const debouncedSearch = useDebounce(searchInput, 300)

  watch(debouncedSearch, (q) => emit('search', q))

  function updateFilter(key: string, value: string) {
    emit('update:modelValue', { ...props.modelValue, [key]: value })
  }
</script>

<template>
  <div class="byz-filter-bar" role="search" :aria-label="'Zoeken en filteren'">
    <input
      v-model="searchInput"
      type="search"
      class="byz-filter-bar__search"
      :placeholder="placeholder"
      :aria-label="placeholder"
    />

    <div class="byz-filter-bar__filters">
      <select
        v-for="filter in filters"
        :key="filter.key"
        class="byz-filter-bar__select"
        :aria-label="filter.label"
        :value="modelValue[filter.key] ?? ''"
        @change="updateFilter(filter.key, ($event.target as HTMLSelectElement).value)"
      >
        <option value="">{{ filter.label }}</option>
        <option
          v-for="opt in filter.options"
          :key="opt"
          :value="opt"
        >{{ opt }}</option>
      </select>
    </div>
  </div>
</template>
```

```vue
<script setup lang="ts">
  import { ref, computed } from 'vue'
  import { useDebounce } from '../../../composables/useDebounce'
  defineOptions({ name: 'ByzMultiSelect' })

  interface Option { value: string; label: string; group?: string }

  interface Props {
    options: Option[]
    modelValue: string[]
    label: string
    placeholder?: string
    disabled?: boolean
  }

  const props = withDefaults(defineProps<Props>(), {
    placeholder: 'Selecteer...',
    disabled: false,
  })

  const emit = defineEmits<{ 'update:modelValue': [v: string[]] }>()

  const open    = ref(false)
  const search  = ref('')
  const debSearch = useDebounce(search, 200)

  const filtered = computed(() =>
    props.options.filter(o =>
      o.label.toLowerCase().includes(debSearch.value.toLowerCase())
    )
  )

  function selectedLabels(values: string[]) {
    return values.map(v => props.options.find(o => o.value === v)?.label ?? v)
  }

  function toggle(value: string) {
    const cur = [...props.modelValue]
    const idx = cur.indexOf(value)
    if (idx >= 0) cur.splice(idx, 1)
    else cur.push(value)
    emit('update:modelValue', cur)
  }

  function remove(value: string) {
    emit('update:modelValue', props.modelValue.filter(v => v !== value))
  }
</script>

<template>
  <div class="byz-multi-select" :class="{ 'byz-multi-select--open': open }">
    <label class="byz-multi-select__label" :id="`byz-ms-label-${label}`">{{ label }}</label>

    <!-- Chips + trigger -->
    <div
      class="byz-multi-select__trigger"
      role="combobox"
      :aria-expanded="open"
      :aria-labelledby="`byz-ms-label-${label}`"
      :aria-haspopup="'listbox'"
      :tabindex="disabled ? -1 : 0"
      @click="open = !open"
      @keydown.enter.prevent="open = !open"
      @keydown.escape="open = false"
    >
      <span
        v-for="val in modelValue"
        :key="val"
        data-chip
        class="byz-multi-select__chip"
      >
        {{ selectedLabels([val])[0] }}
        <button
          type="button"
          data-chip-remove
          class="byz-multi-select__chip-remove"
          :aria-label="`Verwijder ${selectedLabels([val])[0]}`"
          @click.stop="remove(val)"
        >✕</button>
      </span>
      <span v-if="modelValue.length === 0" class="byz-multi-select__placeholder">
        {{ placeholder }}
      </span>
    </div>

    <!-- Dropdown -->
    <div v-if="open" class="byz-multi-select__dropdown">
      <input
        v-model="search"
        type="search"
        class="byz-multi-select__search"
        placeholder="Zoeken..."
        aria-label="Zoeken in opties"
        @click.stop
      />

      <ul role="listbox" class="byz-multi-select__list" :aria-multiselectable="true">
        <li
          v-for="opt in filtered"
          :key="opt.value"
          role="option"
          :data-option="opt.value"
          class="byz-multi-select__option"
          :class="{ 'byz-multi-select__option--selected': modelValue.includes(opt.value) }"
          :aria-selected="modelValue.includes(opt.value)"
          @click.stop="toggle(opt.value)"
        >
          <span aria-hidden="true">{{ modelValue.includes(opt.value) ? '✓' : '○' }}</span>
          {{ opt.label }}
        </li>
      </ul>
    </div>
  </div>
</template>
```

```vue
  interface TreeNode {

    label: string
    icon?: string
    children: TreeNode[]
```
